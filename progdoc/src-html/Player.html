001import java.util.Stack;
002
003/**
004 * Player class
005 *
006 * @author Luca
007 * @version 24/05/2022
008 */
009public class Player
010{
011    private String aPlayerName;
012    private Room aCurrentRoom;
013    private Stack aLastRooms;
014    private ItemList aItems;
015    private int aMaxWeight;
016    private int aWeight;
017    private int aNumberOfMoveLeft;
018
019    /**
020     * Constructor for objects of class Player
021     * @param pCurrentRoom Room
022     */
023    public Player(final Room pCurrentRoom)
024    {
025        this.aPlayerName = "admin";
026        this.aCurrentRoom = pCurrentRoom;
027        this.aLastRooms = new Stack();
028        this.aItems = new ItemList();
029        this.aMaxWeight = 10000;
030        this.aWeight = 0;
031        this.aNumberOfMoveLeft = 50;
032    }//Player()
033
034    /**
035     * Renvoie la Room actuelle
036     * @return Room 
037     */
038    public Room getCurrentRoom() {
039        return this.aCurrentRoom;
040    }//getCurrentRoom()
041    
042    /**
043     * Renvoie les dernières Room
044     * @return Stack 
045     */
046    public Stack getLastRooms() {
047        return this.aLastRooms;
048    }//getLastRooms()
049    
050    /**
051     * Renvoie le nombre de pas restant
052     * @return int Nombre
053     */
054    public int getNumberOfMoveLeft() {
055        return this.aNumberOfMoveLeft;
056    }//getNumberOfMoveLeft()
057    
058    /**
059     * Met la salle actuelle par rapport à la salle en paramètre
060     * @param pRoom Room
061     */
062    public void setRoom(final Room pRoom) {
063        this.aCurrentRoom = pRoom;
064    }//setRoom()
065    
066    /**
067     * Vide les salles enregistrées
068     */
069    public void emptyLastRooms() {
070        this.aLastRooms.clear();
071    }//emptyLastRooms()
072    
073    /**
074     * Change la Room actuelle en la Room où le joueur veut se diriger
075     * @param pDirection Command
076     * @return Room nouvelle, null s'il n'y a pas de porte, Room.UNKNOWN_DIRECTION si la direction que le Player veut prendre est inconnu
077     */
078    public Room goRoom(final Command pDirection) {
079        Room vNextRoom = aCurrentRoom.getExit(pDirection.getSecondWord());
080        if (vNextRoom == Room.UNKNOWN_DIRECTION || vNextRoom == null || vNextRoom == Room.LOCKED_DOOR) {
081            return vNextRoom; 
082        }
083        
084        if (!hasMoveLeft()) {
085            return Room.NO_MOVE_LEFT;
086        }
087        //ajoute la room à la liste des rooms
088        this.aLastRooms.push(this.aCurrentRoom);
089        //met le joueur dans la nouvelle room
090        this.aCurrentRoom = vNextRoom;
091        this.aNumberOfMoveLeft--;
092        
093        return this.aCurrentRoom;
094    }//goRoom()
095    
096    /**
097     * Change la Room actuelle en la dernière Room où le Player s'est déplacé
098     * @return Room
099     */
100    public Room back() {
101        //recup et suppr le dernier objet du stack qu'on transforme en room car on sait que ça en ait un
102        if (!hasMoveLeft()) {
103            return Room.NO_MOVE_LEFT;
104        } else if (!isOpen(getDirection((Room) this.aLastRooms.lastElement()))) {
105            return Room.LOCKED_DOOR;
106        } else {
107             this.aCurrentRoom = (Room) this.aLastRooms.pop();
108             this.aNumberOfMoveLeft--;
109             return this.aCurrentRoom;
110        }
111
112    }//back()
113    
114    /**
115     * Envoie la direction de la salle
116     * @param pRoom Room
117     * @return String 
118     */
119    public String getDirection(final Room pRoom) {
120        return this.aCurrentRoom.getDirection(pRoom);
121    }//getDirection()
122    
123    /**
124     * Envoie si la direction est ouverte ou non
125     * @param pDirection String
126     * @return boolean si ouvert true, false si fermé 
127     */
128    public boolean isOpen(final String pDirection) {
129        Room vNextRoom = aCurrentRoom.getExit(pDirection);
130        return !(vNextRoom == Room.LOCKED_DOOR);
131    }//isOpen()
132    
133    /**
134     * Envoie s'il reste des pas restants
135     * @return boolean vrai s'il en reste, faux s'il n'en reste pas 
136     */
137    private boolean hasMoveLeft() {
138        return !(aNumberOfMoveLeft == 0);
139    }//hasMoveLeft()
140    
141    /**
142     * Ajoute l'Item, que le Player veut récupérer, à son inventaire ainsi que l'enlever de la Room
143     * @param pCommand String
144     * @return String 
145     */
146    public String take(final String pCommand) {
147        Item vItem = this.aCurrentRoom.getItem(pCommand.toLowerCase());
148        if (vItem != null) {
149            if (this.aWeight + vItem.getWeight() <= this.aMaxWeight) {
150                this.addItemPlayer(vItem.getName());
151                this.aCurrentRoom.removeItem(vItem.getName());
152                return vItem.getName().toLowerCase();
153            } else {
154                return "TooMuch";
155            }
156        } else {
157            return null;
158        }
159    }//take()
160    
161    /**
162     * Enlève l'Item, que le joueur veut mettre par terre, de son inventaire ainsi que l'ajouter à la Room
163     * @param pCommand Command
164     * @return String
165     */
166    public String drop(final String pCommand) {
167        if (!this.aItems.isEmpty()) {
168            if (this.aItems.getItem(pCommand) != null) {
169                Item vItem = this.aItems.getItem(pCommand);
170                this.aCurrentRoom.addItem(vItem);
171                this.removeItemPlayer(pCommand);
172                return pCommand.toLowerCase();
173            } else {
174                return null;
175            }
176        } else {
177            return null;
178        }
179        
180    }//drop()
181    
182    /**
183     * Ajoute l'Item à son inventaire ainsi qu'ajouter le poids de l'Item au Player
184     * @param pName de l'Item
185     */
186    public void removeItemPlayer(final String pName) {
187        Item vItem = this.aItems.getItem(pName);
188        this.aItems.removeItem(pName);
189        this.aWeight -= vItem.getWeight();
190    }//removeItemPlayer()
191    
192    /**
193     * Enlève l'Item de son inventaire ainsi qu'enlever le poids de l'Item au Player
194     * @param pName de l'item
195     */
196    public void addItemPlayer(final String pName) {
197        Item vItem = this.aCurrentRoom.getItem(pName);
198        this.aItems.addItem(vItem);
199        this.aWeight += vItem.getWeight();
200    }//addItemPlayer()
201    
202    /**
203     * Retourne l'Item donné par son nom
204     * @param pCommand String
205     * @return Item
206     */
207    public Item getItem(final String pCommand) {
208        return this.aItems.getItem(pCommand);
209    }//getItem()
210    
211    /**
212     * Retourne la description longue de tous les Item de l'inventaire avec le poids du Player et le poids total que le Player peut avoir
213     * @return String
214     */
215    public String inventory() {
216        return this.aItems.getItemString() + "\nYour weight : " + this.aWeight + "/" + this.aMaxWeight + "g.";
217    }//inventory()
218    
219    /**
220     * Mettre le poids maximal au paramètre donné
221     * @param pMaxWeight int
222     */
223    private void setMaxWeight(final int pMaxWeight) {
224        this.aMaxWeight = pMaxWeight;
225    }//setMaxWeight()
226    
227    /**
228     * Mettre le poids maximal fois 2
229     */
230    public void doubleMaxWeight() {
231        this.setMaxWeight(this.aMaxWeight*2);
232    }//doubleMaxWeight()
233}
