002/**
003 * Classe des locked doors. Ceux sont les seules portes qui peuvent être
004 * ouvertes grâce à une clé
005 *
006 * @author Luca
007 * @version 24/05/2022
008 */
009public class LockedDoor extends Door
010{
011    // instance variables - replace the example below with your own
012    private Item aKey;
013
014    /**
015     * Constructor for objects of class Door
016     * @param pIsLocked boolean
017     * @param pKey Item
018     */
019    public LockedDoor(final boolean pIsLocked, final Item pKey) {
020        super(pIsLocked);
021        this.aKey = pKey;
022    }
023
024    /**
025     * Accesseur de l'Item demandée pour ouvrir et fermer la porte
026     * @return Item 
027     */
028    public Item getKey() {
029        return this.aKey;
030    }//getKey()
031}
