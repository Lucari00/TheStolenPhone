001/**
002 * Representations for all the valid command words for the game.
003 * 
004 * @author Luca
005 * @version 24/05/2022
006 */
007public enum CommandWord {
008    // A value for each command word, plus one for unrecognised
009    // commands.
010    GO("go", "aller"), QUIT("quit", "quitter"), HELP("help", "aide"), UNKNOWN("?", "?"),
011    LOOK("look", "regarder"), EAT("eat", "manger"), BACK("back", "retour"),
012    TEST("test", "test"), TAKE("take", "prendre"), DROP("drop", "jeter"), 
013    INVENTORY("inventory", "inventaire"), ROOMIS("roomIs", "roomIs"), 
014    ROOMHAS("roomHas", "roomHas"), ROOMHASNOT("roomHasNot", "roomHasNot"), 
015    PLAYERHAS("playerHas", "playerHas"), PLAYERHASNOT("playerHasNot", "playerHasNot"),
016    CHARGE("charge", "charger"), FIRE("fire", "tirer"), OPEN("open", "ouvrir"),
017    CLOSE("close","fermer"), ALEA("alea", "alea"), TESTMOD("testmod", "testmod"),
018    ;
019    
020    private String aEn; //command en anglais
021    private String aFr; //command en fran√ßais
022    
023    /**
024     * Initialise with the corresponding command word and the translate in fr.
025     * @param pEn The command string 
026     * @param pFr The command string in Fr.
027     */
028    CommandWord(final String pEn, final String pFr) {
029        this.aEn = pEn;
030        this.aFr = pFr;
031    }//CommandWord()
032    
033    /**
034     * @return The command word as a string in english.
035     */
036    public String getEnWord()
037    {
038        return aEn;
039    }//getEnWord()
040    
041    /**
042     * @return The command word as a string in french.
043     */
044    public String getFrWord()
045    {
046        return aFr;
047    }//getFrWord()
048}
