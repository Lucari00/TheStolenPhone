007/**

013public class TakeCommand extends Command
014{
015    
016    /**
017     * Constructor for objects of class GoCommand
018     */
019    public TakeCommand()
020    {
021    }
022
023    /** 
024     * Fonction de la commande go
025     * @param aPlayer Player
026     * @param aGui UserInterface
027     * @param pGameEngine GameEngine
028     * 
029     * return boolean vrai pour arrÃªter le jeu, faux pour continuer
030     */
031    public boolean execute(Player aPlayer, UserInterface aGui, GameEngine pGameEngine) {
032        if (hasSecondWord()) {
033            String vItemName = aPlayer.take(getSecondWord());
034            if (vItemName != null && !vItemName.equals("TooMuch")) {
035                aGui.println("You took " + vItemName + " !");
036            } else if (vItemName != null && vItemName.equals("TooMuch")) {
037                aGui.println("You are too heavy, you can't take that item. You can try to drop some items if you want it.");
038            } else {
039                aGui.println("This item is not in the room.");
040            }
041        } else {
042            aGui.println("take what ?");
043        }
044        return false;
045    }
046}
