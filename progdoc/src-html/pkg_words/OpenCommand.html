001package pkg_words;
002
003import pkg_characters.Player;
004import pkg_engine.UserInterface;
005import pkg_engine.GameEngine;
006import pkg_doors.Door;
007import pkg_doors.LockedDoor;
008
009/**
010 * Classe de la commande OpenCommando
011 *
012 * @author Luca
013 * @version 24/05
014 */
015public class OpenCommand extends Command
016{
017    
018    /**
019     * Constructor for objects of class OpenCommand
020     */
021    public OpenCommand()
022    {
023    }
024
025    /** 
026     * Fonction de la commande OpenCommand
027     * @param aPlayer Player
028     * @param aGui UserInterface
029     * @param pGameEngine GameEngine
030     * 
031     * return boolean vrai pour arrêter le jeu, faux pour continuer
032     */
033    public boolean execute(Player aPlayer, UserInterface aGui, GameEngine pGameEngine) {
034        if (!hasSecondWord()) {
035            aGui.println("I need to know which door to open.");
036            return false;
037        }
038        
039        //récupère la salle actuelle, puis récupère les portes, pour récupérer celle de la direction qu'on cherche
040        Object vObj = aPlayer.getCurrentRoom().getDoor(getSecondWord().toLowerCase());
041        
042        Door vDoor = (Door) vObj;
043        if (vDoor == Door.UNKNOWN_DIRECTION_DOOR) {
044            aGui.println( "Unknown direction !" );
045            return false;
046        } else if (vDoor == null) {
047            aGui.println( "There is no door !" );
048            return false;
049        } else if (vDoor.isLocked()) {
050            //si le joueur a la clé sur lui
051            try {
052                LockedDoor vLockedDoor = (LockedDoor) vObj;
053                if (aPlayer.getItem(vLockedDoor.getKey().getName()) == vLockedDoor.getKey()) {
054                vDoor.setLocked(false);
055                aGui.println( "The door has been unlocked !" );
056            } else {
057                aGui.println( "You need to have the key to open the door." );
058            }
059            } catch (ClassCastException ex) {
060                aGui.println( "You don't have the key to locked." );
061            }
062            return false;
063        } else {
064            aGui.println( "The door is already open !" );
065            return false;
066        }
067    }
068}
