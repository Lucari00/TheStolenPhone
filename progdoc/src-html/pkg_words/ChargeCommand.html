008/**
009 * Classe de la commande ChargeCommand

011 * @author Luca

013 */
014public class ChargeCommand extends Command
015{
016    
017    /**
018     * Constructor for objects of class ChargeCommand
019     */
020    public ChargeCommand()
021    {
022    }
023
024    /** 
025     * Fonction de la commande ChargeCommand
026     * @param aPlayer Player
027     * @param aGui UserInterface
028     * @param pGameEngine GameEngine
029     * 
030     * return boolean vrai pour arrÃªter le jeu, faux pour continuer
031     */
032    public boolean execute(Player aPlayer, UserInterface aGui, GameEngine pGameEngine) {
033        if (hasSecondWord()) {
034            aGui.println("I don't know how to charge something in particular.");
035            return false;
036        }
037        
038        Beamer vB = (Beamer) aPlayer.getItem("beamer");
039        if (vB != null) {
040            vB.charge(aPlayer.getCurrentRoom());
041            aGui.println("You just charge this room, type 'fire' to teleport to it when you want.");
042        } else {
043            aGui.println("You don't have a beamer in your inventory.");
044        }
045        return false;
046    }
047}
