008/**
009 * Classe de la commande DropCommand
010 *
011 * @author Luca
012 * @version 24/05
013 */
014public class DropCommand extends Command
015{
016    
017    /**
018     * Constructor for objects of class DropCommand
019     */
020    public DropCommand()
021    {
022    }
023
024    /** 
025     * Fonction de la commande DropCommand
026     * @param aPlayer Player
027     * @param aGui UserInterface
028     * @param pGameEngine GameEngine
029     * 
030     * return boolean vrai pour arrÃªter le jeu, faux pour continuer
031     */
032    public boolean execute(Player aPlayer, UserInterface aGui, GameEngine pGameEngine) {
033        if (hasSecondWord()) {
034            String vItemName = aPlayer.drop(getSecondWord());
035            if (vItemName.toLowerCase().equals("phone") && aPlayer.getCurrentRoom().getImageName().toLowerCase().equals("houseinside.jpg")) {
036                aGui.println("Wow you win this game !");
037                aGui.println("BRAVO!");
038                aGui.enable( false );
039                Room vWin = Room.WIN;
040                if ( vWin.getImageName() != null )
041                    aGui.showImage( vWin.getImageName() );
042            } else if (vItemName != null) {
043                aGui.println("You drop " + vItemName + ".");
044            } else {
045                aGui.println("You don't have this item in you inventory");
046            }
047        } else {
048            aGui.println("drop what ?");
049        }
050        return false;
051    }
052}
