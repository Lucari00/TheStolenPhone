001package pkg_words;
002
003import java.util.HashMap;
004import pkg_words.Command;
005import java.util.ArrayList;
006
007/**
008 * This class is part of the "World of Zuul" application. 
009 * "World of Zuul" is a very simple, text based adventure game.  
010 * 
011 * This class holds an enumeration table of all command words known to the game.
012 * It is used to recognise commands as they are typed in.
013 *
014 * @author  Michael Kolling and David J. Barnes + D.Bureau
015 * @version 2008.03.30 + 2019.09.25 + 24/05/2022
016 */
017public class CommandWords
018{
019    // an HashMap that will hold all valid command words
020    private HashMap<String, CommandWord> pValidCommands;
021    private HashMap<CommandWord, Command> aCommands;
022
023    /** 
024     * Constructor - initialise the command words.
025     */
026    public CommandWords() {
027        this.pValidCommands = new HashMap<String, CommandWord>();
028        for(CommandWord vCommand : CommandWord.values()) {
029            if(vCommand != CommandWord.UNKNOWN) {
030                pValidCommands.put(vCommand.getEnWord(), vCommand);
031                pValidCommands.put(vCommand.getFrWord(), vCommand);
032            } 
033        }
034        
035        CommandWord[] vCom = CommandWord.values();
036
037        this.aCommands = new HashMap<CommandWord, Command>();
038        this.aCommands.put(vCom[0], new GoCommand());
039        this.aCommands.put(vCom[2], new HelpCommand(this.getCommandList())); 
040        this.aCommands.put(vCom[1], new QuitCommand());
041        this.aCommands.put(vCom[4], new LookCommand());
042        this.aCommands.put(vCom[5], new EatCommand());
043        this.aCommands.put(vCom[6], new BackCommand());
044        this.aCommands.put(vCom[7], new TestCommand());
045        this.aCommands.put(vCom[8], new TakeCommand());
046        this.aCommands.put(vCom[9], new DropCommand());
047        this.aCommands.put(vCom[10], new InventoryCommand());
048        this.aCommands.put(vCom[11], new RoomIsCommand());
049        this.aCommands.put(vCom[12], new RoomHasCommand());
050        this.aCommands.put(vCom[13], new RoomHasNotCommand());
051        this.aCommands.put(vCom[14], new PlayerHasCommand());
052        this.aCommands.put(vCom[15], new PlayerHasNotCommand());
053        this.aCommands.put(vCom[16], new ChargeCommand());
054        this.aCommands.put(vCom[17], new FireCommand());
055        this.aCommands.put(vCom[18], new OpenCommand());
056        this.aCommands.put(vCom[19], new CloseCommand());
057        this.aCommands.put(vCom[20], new AleaCommand());
058        this.aCommands.put(vCom[21], new TestModCommand());
059        this.aCommands.put(vCom[22], new GiveCommand());
060        this.aCommands.put(vCom[23], new FightCommand());
061    } // CommandWords()
062
063    /**
064     * Check whether a given String is a valid command word.
065     * @param pString String
066     * @return true if a given string is a valid command,
067     * false if it isn't.
068     */
069    public boolean isCommand( final String pString )
070    {
071        return pValidCommands.containsKey(pString);
072    } // isCommand()
073    
074    public Command get(String pWord) {
075        CommandWord vP = this.pValidCommands.get(pWord);
076        return (Command)this.aCommands.get(vP);
077    }
078    
079    /**
080     * Find the CommandWord associated with a command word.
081     * @param pCommandWord The word to look up.
082     * @return The CommandWord correspondng to commandWord, or UNKNOWN
083     *         if it is not a valid command word.
084     */
085    public CommandWord getCommandWord(final String pCommandWord) {
086        CommandWord vCommand = pValidCommands.get(pCommandWord);
087        if (vCommand != null) {
088            return vCommand;
089        } else {
090            return CommandWord.UNKNOWN;
091        }
092    }
093    
094    /**
095     * 
096     * @return String of all commands
097     */
098    public String getCommandList() {
099        StringBuilder vText = new StringBuilder();
100        vText.append(" ");
101        for (String vCommand : pValidCommands.keySet()) {
102            if (vCommand.equals("roomHas") || vCommand.equals("roomHasNot") || vCommand.equals("playerHas") || vCommand.equals("playerHasNot") || vCommand.equals("roomIs")) {
103                continue;
104            }
105            vText.append(vCommand + " ");
106        }
107        return vText.toString();
108    }
109} // CommandWords
