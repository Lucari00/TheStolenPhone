001package pkg_words;

003/**
004 * Representations for all the valid command words for the game.
005 * 
006 * @author Luca
007 * @version 24/05/2022
008 */
009public enum CommandWord {
010    // A value for each command word, plus one for unrecognised
011    // commands.
012    GO("go", "aller"), QUIT("quit", "quitter"), HELP("help", "aide"), UNKNOWN("?", "?"),
013    LOOK("look", "regarder"), EAT("eat", "manger"), BACK("back", "retour"),
014    TEST("test", "test"), TAKE("take", "prendre"), DROP("drop", "jeter"), 
015    INVENTORY("inventory", "inventaire"), ROOMIS("roomIs", "roomIs"), 
016    ROOMHAS("roomHas", "roomHas"), ROOMHASNOT("roomHasNot", "roomHasNot"), 
017    PLAYERHAS("playerHas", "playerHas"), PLAYERHASNOT("playerHasNot", "playerHasNot"),
018    CHARGE("charge", "charger"), FIRE("fire", "tirer"), OPEN("open", "ouvrir"),
019    CLOSE("close","fermer"), ALEA("alea", "alea"), TESTMOD("testmod", "testmod"),
020    GIVE("give", "donner"), FIGHT("fight", "battre");
021    
022    private String aEn; //command en anglais
023    private String aFr; //command en fran√ßais
024    
025    /**
026     * Initialise with the corresponding command word and the translate in fr.
027     * @param pEn The command string 
028     * @param pFr The command string in Fr.
029     */
030    CommandWord(final String pEn, final String pFr) {
031        this.aEn = pEn;
032        this.aFr = pFr;
033    }//CommandWord()
034    
035    /**
036     * @return The command word as a string in english.
037     */
038    public String getEnWord()
039    {
040        return aEn;
041    }//getEnWord()
042    
043    /**
044     * @return The command word as a string in french.
045     */
046    public String getFrWord()
047    {
048        return aFr;
049    }//getFrWord()
050}
