001package pkg_words;
002
003import pkg_characters.Player;
004import pkg_engine.UserInterface;
005import pkg_engine.GameEngine;
006import pkg_characters.Character;
007import pkg_characters.MovingCharacter;
008import pkg_items.Item;
009import pkg_doors.Door;
010
011/**
012 * Classe de la commande GiveCommand
013 *
014 * @author Luca
015 * @version 24/05
016 */
017public class GiveCommand extends Command
018{
019    
020    /**
021     * Constructor for objects of class GiveCommand
022     */
023    public GiveCommand()
024    {
025    }
026
027    /** 
028     * Fonction de la commande GiveCommand
029     * @param aPlayer Player
030     * @param aGui UserInterface
031     * @param pGameEngine GameEngine
032     * 
033     * return boolean vrai pour arrêter le jeu, faux pour continuer
034     */
035    public boolean execute(Player aPlayer, UserInterface aGui, GameEngine pGameEngine) {
036        if (hasSecondWord()) {
037            aGui.println( "You can't have two or more words" );
038            return false;
039        }
040        
041        Character vChar = aPlayer.getCurrentRoom().getCharacter();
042        if (vChar != null) {
043            Item vCharItem = vChar.getItemNeed();
044            if (vCharItem != null) {
045                if (aPlayer.getItem(vCharItem.getName()) != null) {
046                    aPlayer.removeItemPlayer(vCharItem.getName());
047                    aPlayer.addItem(vChar.getItemGive());
048                    aGui.println(vChar.getDialGive());
049                    
050                    if (vCharItem.getName().equals("KeyFBI")) { //si c'est le pnj du fbi
051                        Door vDoor = (Door) pGameEngine.getRoom(4).getDoor("up"); //récupère la porte du fbi et la déverouille
052                        vDoor.setLocked(false);
053                        aGui.println( "The building of FBI is unlocked !" );
054                        
055                        vChar.clearDialogue();
056                        vChar.addDialogue("Thanks you so much for helping me, come to the office when you want !");
057                        MovingCharacter vMovChar = (MovingCharacter) vChar;
058                        vMovChar.addRoom(pGameEngine.getRoom(5)); 
059                    }
060                    
061                    vChar.setItemNeed(null);
062                    vChar.setItemGive(null);   
063                } else {
064                    aGui.println( "You don't have the item needed." );
065                }
066            } else {
067                aGui.println( "There is nothing to give here." );
068            }
069        } else {
070            aGui.println( "There is nothing to give here." );
071        }
072        return false;
073    }
074}
