014public class AleaCommand extends Command
015{
016    
017    /**
018     * Constructor for objects of class AleaCommand

020    public AleaCommand()
021    {
022    }
023
024    /** 
025     * Fonction de la commande AleaCommand
026     * @param aPlayer Player
027     * @param aGui UserInterface
028     * @param pGameEngine GameEngine
029     * 
030     * return boolean vrai pour arrÃªter le jeu, faux pour continuer
031     */
032    public boolean execute(Player aPlayer, UserInterface aGui, GameEngine pGameEngine) {
033        if (pGameEngine.getTestMod()) {
034            TransporterRoom vRoomT = pGameEngine.getTransR(); 
035            if (hasSecondWord()) {
036                try {
037                    int vNum = Integer.parseInt(getSecondWord());
038                    if (vNum > 0 && vNum < pGameEngine.getRoomsSize()) {
039                        vRoomT.setAlea(vNum);
040                        aGui.println("Succesful.");
041                    } else {
042                        aGui.println("Your number must be between 0 and the number of room (" + pGameEngine.getRoomsSize() + ", not included).");
043                    } 
044                } catch (final NumberFormatException NFE) {
045                    aGui.println("After alea, you need to write a number");
046                }
047            } else {
048                vRoomT.setAlea(-1); 
049            }
050        } else {
051            aGui.println( "That's cheating !!!!!!!!!!!" );
052        }
053        return false;
054    }
055}
