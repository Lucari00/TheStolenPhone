001package pkg_words;
002
003import pkg_characters.Player;
004import pkg_engine.UserInterface;
005import pkg_engine.GameEngine;
006import pkg_rooms.Room;
007
008/**
009 * Classe de la commande go
010 *
011 * @author Luca
012 * @version 24/05
013 */
014public class GoCommand extends Command
015{
016    /**
017     * Constructor for objects of class GoCommand
018     */
019    public GoCommand()
020    {
021    }
022
023    /** 
024     * Fonction de la commande PlayerHasNotCommand
025     * @param aPlayer Player
026     * @param aGui UserInterface
027     * @param pGameEngine GameEngine
028     * 
029     * return boolean vrai pour arrÃªter le jeu, faux pour continuer
030     */
031    public boolean execute(Player aPlayer, UserInterface aGui, GameEngine pGameEngine)
032    {
033        //pas de second mot
034        if (!hasSecondWord()) {
035            aGui.println("Go where ?");
036            return false;
037        }
038        
039        //next Room ?
040        Room vNextRoom = aPlayer.goRoom(this.getSecondWord()); 
041        
042        if (vNextRoom == Room.UNKNOWN_DIRECTION) {
043            aGui.println( "Unknown direction !" );
044            return false;
045        } else if (vNextRoom == null) {
046            aGui.println( "There is no door !" );
047            return false;
048        } else if (vNextRoom == Room.NO_MOVE_LEFT) {
049            aGui.println("You are too tired, you can't move anymore.");
050            aGui.println("Game over");
051            aGui.enable( false );
052            if ( vNextRoom.getImageName() != null )
053                aGui.showImage( vNextRoom.getImageName() );
054            return false;
055        } else if (vNextRoom == Room.LOCKED_DOOR) {
056            aGui.println( "The door is locked !" );
057            return false;
058        } else if (vNextRoom.isTransporter()) {
059            aPlayer.emptyLastRooms();
060        }
061        
062        //move pnj
063        pGameEngine.moveCharacter();
064        //affiche la description et les directions possibles
065        pGameEngine.showInterface(vNextRoom);
066        return false;
067    }
068}
