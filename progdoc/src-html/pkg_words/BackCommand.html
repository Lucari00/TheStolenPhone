001package pkg_words;
002
003import pkg_characters.Player;
004import pkg_engine.UserInterface;
005import pkg_engine.GameEngine;
006import pkg_rooms.Room;
007
008/**
009 * Classe de la commande go
010 *
011 * @author Luca
012 * @version 24/05
013 */
014public class BackCommand extends Command
015{
016    
017    /**
018     * Constructor for objects of class BackCommand
019     */
020    public BackCommand()
021    {
022    }
023
024    /** 
025     * Fonction de la commande BackCommand
026     * @param aPlayer Player
027     * @param aGui UserInterface
028     * @param pGameEngine GameEngine
029     * 
030     * return boolean vrai pour arrÃªter le jeu, faux pour continuer
031     */
032    public boolean execute(Player aPlayer, UserInterface aGui, GameEngine pGameEngine) {
033        if (hasSecondWord()) {
034            aGui.println("I don't know how to go back in a particular way.");
035            return false;
036        } else if (aPlayer.getLastRooms().empty()) {
037            aGui.println("You didn't move back further, so you can't go back.");
038            return false;
039        }
040        
041        Room vCurrentRoom = aPlayer.back();
042        
043        if (vCurrentRoom == Room.NO_MOVE_LEFT) {
044            aGui.println("You are too tired, you can't move anymore.");
045            aGui.println("Game over");
046            aGui.enable( false );
047        } else if (vCurrentRoom == Room.LOCKED_DOOR) {
048            aGui.println( "The door is locked !" );
049            return false;
050        }
051        
052        pGameEngine.moveCharacter();
053        pGameEngine.showInterface(vCurrentRoom);
054        return false;
055    }
056}
