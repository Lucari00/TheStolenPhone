001package pkg_characters;
002
003import pkg_rooms.Room;
004import java.util.ArrayList;
005import pkg_items.Item;
006
007/**
008 * Classe d'un character
009 *
010 * @author Luca
011 * @version 24/05/2022
012 */
013public class Character
014{
015    private String aName;
016    private ArrayList<String> aDialogue;
017    private String aDialGive;
018    private Room aCurrentRoom;
019    private int aCompteur;
020    private Item aNeedItem; //Item qu'il veut
021    private Item aGiveItem; //Item qu'il donne
022    
023    /**
024     * Constructor for objects of class Character
025     * @param pName String
026     * @param pCurrentRoom Room
027     * @param pNeedItem Item
028     * @param pGiveItem Item
029     */
030    public Character(final String pName, final Room pCurrentRoom, final Item pNeedItem, final Item pGiveItem) {
031        this.aName = pName;
032        this.aCurrentRoom = pCurrentRoom;
033        this.aDialogue = new ArrayList<String>();
034        this.aCompteur = 0;
035        this.aNeedItem = pNeedItem;
036        this.aGiveItem = pGiveItem;
037    }
038    
039    /**
040     * Getter du nom du PNJ
041     * @return String
042     */
043    public String getName() {
044        return this.aName;
045    }
046    
047    /**
048     * Setter de la room actuelle du PNJ
049     * @param pRoom Room
050     */
051    public void setRoom(final Room pRoom) {
052        this.aCurrentRoom = pRoom;
053    }
054    
055    /**
056     * Setter de la ligne de dialogue quand le personnage donne l'objet
057     * @param pS String
058     */
059    public void setDialGive(final String pS) {
060        this.aDialGive = pS;
061    }
062    
063    /**
064     * Getter de la ligne de dialogue
065     * @return String
066     */
067    public String getDialGive() {
068        return this.aDialGive;
069    }
070    
071    /**
072     * Ajout d'une ligne de dialogue
073     * @param pS String
074     */
075    public void addDialogue(final String pS) {
076        this.aDialogue.add(pS);
077    }
078    
079    /**
080     * Getter d'une ligne de dialogue
081     * @return String
082     */
083    public String getDialogue() {
084        String vText = this.aName + " : " + this.aDialogue.get(this.aCompteur);
085        
086        this.aCompteur++;
087        
088        if (this.aCompteur == this.aDialogue.size()) {
089            this.aCompteur = 0;
090        }
091
092        return vText;
093    }
094    
095    /**
096     * Getter de la room actuelle
097     * @return Room
098     */
099    public Room getCurrentRoom() {
100        return this.aCurrentRoom;
101    }
102    
103    /**
104     * Getter de l'item demandé
105     * @return Item
106     */
107    public Item getItemNeed() {
108        return this.aNeedItem;
109    }
110    
111    /**
112     * Getter de l'item donné
113     * @return Item
114     */
115    public Item getItemGive() {
116        return this.aGiveItem;
117    }
118    
119    /**
120     * Setter de l'item demandé
121     * @param pItem Item
122     */
123    public void setItemNeed(final Item pItem) {
124        this.aNeedItem = pItem;
125    }
126    
127    /**
128     * Setter de l'item donné
129     * @param pItem Item
130     */
131    public void setItemGive(final Item pItem) {
132        this.aGiveItem = pItem;
133    }
134    
135    /**
136     * Clear toutes les lignes de dialogue
137     */
138    public void clearDialogue() {
139        this.aDialogue.clear();
140        this.aCompteur = 0;
141    }
142}
