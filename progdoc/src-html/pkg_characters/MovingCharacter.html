001package pkg_characters;
002
003import pkg_rooms.Room;
004import java.util.ArrayList;
005import pkg_items.Item;
006
007/**
008 * Classe d'un character
009 *
010 * @author Luca
011 * @version 24/05/2022
012 */
013public class MovingCharacter extends Character
014{
015    private ArrayList<Room> aRooms;
016    private int aRoomComp;
017    
018    /**
019     * Constructor for objects of class MovingCharacter
020     * @param pName String
021     * @param pCurrentRoom Room
022     * @param pNeedItem Item
023     * @param pGiveItem Item
024     */
025    public MovingCharacter(final String pName, final Room pCurrentRoom, final Item pNeedItem, final Item pGiveItem) {
026        super(pName, pCurrentRoom, pNeedItem, pGiveItem);
027        this.aRooms = new ArrayList<Room>();
028    }
029    
030    /**
031     * Ajout d'une room à la liste des rooms où le pnj se déplace
032     * @param pRoom Room
033     */
034    public void addRoom(final Room pRoom) {
035        this.aRooms.add(pRoom);
036    }
037    
038    /**
039     * Méthode qui fait avancer le pnj dans la pièce suivante
040     */
041    public void changeRoom() {
042        this.aRoomComp++;
043        
044        if (this.aRoomComp == this.aRooms.size()) {
045            this.aRoomComp = 0;
046        }
047        
048        this.getCurrentRoom().setCharacter(null);
049        this.setRoom(this.aRooms.get(this.aRoomComp));
050        this.getCurrentRoom().setCharacter(this);
051    }
052}
