292    /**
293     * Détermine la commande passée et l'exécute
294     * @param  pCommandLine String général
295     * @return    True pour quitter le jeu, false si la Commande est pour autre chose
296     */
297    public boolean processCommand(final String pCommandLine) {
298        this.aGui.println( "> " + pCommandLine );
299        Command vCommand = this.aParser.getCommand(pCommandLine); 
300        
301        if (vCommand == null) {
302            this.aGui.println("I don't know what you mean...");
303            return false;
304        } else {
305            if (vCommand.execute(this.aPlayer, this.aGui, this)) {
306                this.endGame();
307                return true;
308            } else {
309                return false;
310            }
311        }
312    }// processCommand()
