001package pkg_items;
002
003/**
004 * Classe qui définie un item
005 *
006 * @author Luca
007 * @version 13/04/22
008 */
009public class Item
010{
011    private String aName;
012    private String aDescription;
013    private int aWeight;
014    private int aPrice;
015
016    /**
017     * Constructor for objects of class Item
018     * @param pName String
019     * @param pDescription String 
020     * @param pWeight int 
021     * @param pPrice int
022     */
023  public Item(final String pName,  final String pDescription, final int pWeight, final int pPrice)
024    {
025        this.aName = pName;
026        this.aDescription = pDescription;
027        this.aWeight = pWeight;
028        this.aPrice = pPrice;
029    }//Item()
030
031    /**
032     * Envoie la description de l'item
033     * @return String nom et la description
034     */
035    public String getDescription()
036    {
037        return this.aName + " : " + this.aDescription;
038    }//getDescription()
039    
040    /**
041     * Envoie le poids de l'item
042     * @return int Poids
043     */
044    public int getWeight() {
045        return this.aWeight;
046    }//getWeight()
047    
048    /**
049     * Envoie le prix de l'item
050     * @return int Prix
051     */
052    public int getPrice() {
053        return this.aPrice;
054    }
055    
056    /**
057     * Envoie le nom de l'item
058     * @return String Nom
059     */
060    public String getName() {
061        return this.aName;
062    }//getName()
063    
064    /**
065     * Envoie la description avec le nom, la description, le poids et le prix
066     * @return String long description
067     */
068    public String getLongDescription() {
069       return this.aName + " : " + this.aDescription + " His weight : " + this.aWeight + "g. His price : " + this.aPrice + "€.";
070    }//getLongDescription()
071    
072}//Item
