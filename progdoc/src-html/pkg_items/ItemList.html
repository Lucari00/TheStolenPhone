001package pkg_items;
002
003import java.util.HashMap;
004import java.util.Set;
005import java.util.Iterator;
006
007/**
008 * Gestion des items
009 *
010 * @author Luca
011 * @version 12/04/2022
012 */
013public class ItemList
014{
015    private HashMap<String, Item> aItems;
016
017    /**
018     * Constructor de ItemList
019     */
020    public ItemList() {
021        this.aItems = new HashMap<String, Item>();
022    }//ItemList()
023    
024    /**
025     * Retourne l'item demandé
026     * @param pName String de l'item
027     * @return Item 
028     */
029    public Item getItem(final String pName) {
030        return this.aItems.get(pName.toLowerCase());
031    }//getItem()
032    
033    /**
034     * Retourne si la HashMap est vide ou non
035     * @return boolean, vrai si vide, faux si non vide
036     */
037    public boolean isEmpty() {
038        return this.aItems.isEmpty();
039    }//isEmpty()
040    
041    /**
042     * Ajoute l'item à la HashMap
043     * @param pItem Item
044     */
045    public void addItem(final Item pItem) {
046        this.aItems.put(pItem.getName().toLowerCase(), pItem);
047    }//addItem()
048    
049    /**
050     * Enlève l'item de la HashMap
051     * @param pName String de l'item
052     */
053    public void removeItem(final String pName) {
054        this.aItems.remove(pName.toLowerCase());
055    }//removeItem()
056    
057    /**
058     * Envoie la description avec le nom, la description, le poids et le prix
059     * @return String
060     */
061    public String getItemString() {
062        StringBuilder vText = new StringBuilder("Items:\n");
063        if (!this.aItems.isEmpty()) {
064            Iterator<Item> vIt = this.aItems.values().iterator();
065            while (vIt.hasNext()) {
066                vText.append(vIt.next().getDescription());
067                if (!vIt.hasNext()) {
068                    break;
069                }
070                vText.append("\n");
071            }
072        } else {
073            return "No item.";
074        }
075        
076        return vText.toString();
077    }//getItemString()
078}
