002/**
003 * Classe de l'item beamer
004 *
005 * @author Luca
006 * @version 24/05/2022
007 */
008public class Beamer extends Item
009{
010    private Room aChargedRoom;
011
012    /**
013     * Constructor for objects of class Beamer
014     * @param pName String
015     * @param pDescription String
016     * @param pWeight int
017     * @param pPrice int
018     */
019    public Beamer(final String pName,final String pDescription, final int pWeight, final int pPrice) {
020        super(pName, pDescription, pWeight, pPrice);
021        this.aChargedRoom = null;
022    }//Beamer()
023    
024    /**
025     * Accesseur si le beamer est chargé
026     * @return boolean
027     */
028    public boolean isCharged() {
029        return this.aChargedRoom != null;
030    }//isCharged()
031    
032    /**
033     * Commande pour charger le beamer de la salle
034     * @param pRoom Room de la salle à charger
035     */
036    public void charge(final Room pRoom) {
037        this.aChargedRoom = pRoom;
038    }//charge()
039    
040    /**
041     * Accesseur de la salle chargée
042     * @return Room
043     */
044    public Room getChargedRoom() {
045        return this.aChargedRoom;
046    }//getChargedRoom()
047}
