002/**
003 * Classe d'une porte
004 *
005 * @author Luca
006 * @version 24/05/2022
007 */
008public class Door
009{
010    private boolean aIsLocked;
011    
012    public static final Door UNKNOWN_DIRECTION_DOOR = new Door(false);
013
014    /**
015     * Constructor for objects of class Door
016     * @param pIsLocked boolean
017     */
018    public Door(final boolean pIsLocked) {
019        this.aIsLocked = pIsLocked;
020        
021    }//Door()
022    
023    /**
024     * Accesseur de si la porte est fermée ou non
025     * @return boolean vrai si fermée, faux si ouvert
026     */
027    public boolean isLocked() {
028        return aIsLocked;
029    }//isLocked()
030    
031    /**
032     * Getter pour changer l'état de la porte, ouvert ou fermée
033     * @param pC boolean, true fermée, false ouvert
034     */
035    public void setLocked(final boolean pC) {
036        this.aIsLocked = pC;
037    }//setLocked()
038}
