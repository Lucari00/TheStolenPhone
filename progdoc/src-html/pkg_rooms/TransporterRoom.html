001package pkg_rooms;
002
003import java.util.ArrayList;
004
005/**
006 * Classe de Transporter Room
007 *
008 * @author Luca
009 * @version 24/05/2022
010 */
011public class TransporterRoom extends Room
012{
013    private RoomRandomizer aRoomR;
014    
015    /**
016     * Constructeur d'une room
017     * @param pDescription String
018     * @param pImage String
019     * @param pRooms ArrayList de Room
020     */
021    public TransporterRoom(final String pDescription, final String pImage, final ArrayList<Room> pRooms) {
022        super(pDescription, pImage);
023        this.aRoomR = new RoomRandomizer(pRooms); 
024    }
025
026    /**
027     * renvoie la sortie qui est donnée par RoomRandomizer
028     * @param pDirection String
029     * @return Room
030     */
031    @Override
032    public Room getExit(final String pDirection) {
033        return this.aRoomR.findRandomRoom();
034    }
035    
036    /**
037     * renvoie si la room est un transporter 
038     * @return boolean vrai
039     */
040    @Override
041    public boolean isTransporter() {
042        return true;
043    }
044    
045    /**
046     * met l'alea à une salle précise
047     * @param pNum int de l'alea donc de la room de l'ArrayList
048     */
049    public void setAlea(final int pNum) {
050        this.aRoomR.setAlea(pNum);
051    }
052}
