001package pkg_rooms;
002
003import java.util.HashMap;
004import java.util.Set;
005import pkg_items.Item;
006import pkg_items.ItemList;
007import pkg_doors.Door;
008import pkg_characters.Character;
009
010/**
011 * Classe qui permet de créer et gérer les salles du jeu
012 *
013 * @author PALAYSI Luca
014 * @version 24/05/2022
015 */
016public class Room
017{
018    private String aDescription;
019    private HashMap<String, Room> aExits;
020    private HashMap<String, Door> aDoors;
021    public static final Room UNKNOWN_DIRECTION = new Room( "nowhere", "" );
022    public static final Room NO_MOVE_LEFT = new Room( "nomove", "GameOver.jpg" );
023    public static final Room WIN = new Room( "win", "Win.jpg" );
024    public static final String[] KNOWN_DIRECTION = {"north", "east", "south", "west", "up", "down"};
025    public static final Room LOCKED_DOOR = new Room( "lockeddoor", "" );
026    private String aImageName;
027    private ItemList aItems;
028    private Character aCharacter;
029
030    /**
031     * Constructeur d'une room
032     * @param pDescription String
033     * @param pImage String
034     */
035    public Room(final String pDescription, final String pImage) {
036        this.aDescription = pDescription;
037        this.aExits = new HashMap<String, Room>();
038        this.aDoors = new HashMap<String, Door>();
039        this.aImageName = pImage;
040        this.aItems = new ItemList();
041        this.aCharacter = null;
042    }// Room()
043    
044    public void setCharacter(final Character pCharacter) {
045        this.aCharacter = pCharacter;
046    }
047    
048    public Character getCharacter() {
049        return this.aCharacter;
050    }
051    
052    /**
053     * Accesseur de la description
054     * @return shortDescription
055     */
056    public String getShortDescription() {
057        return this.aDescription;
058    }//getDescription()
059    
060    /**
061     * Accesseur de la description avec la sortie
062     * @return String longDescription
063     */
064    public String getLongDescription() {
065        return "You are " + this.aDescription + "\n" + this.aItems.getItemString() + this.getStringChar() +"\n" + this.getExitString();
066    }//getLongDescription()
067    
068    /**
069     * Accesseur sorties
070     * @param  pDirection String de la direction souhaitée
071     * @return  Room  La salle de la direction, null s'il n'y pas de salle, unknown direction si la direction est inconnue
072     */
073    public Room getExit(final String pDirection) {
074        for (int i = 0; i < KNOWN_DIRECTION.length; i++) {
075            if (KNOWN_DIRECTION[i].equals(pDirection)) {
076                if (this.aDoors.get(pDirection) != null) {
077                    if (!this.aDoors.get(pDirection).isLocked())
078                        return this.aExits.get(pDirection);
079                    else 
080                        return LOCKED_DOOR;
081                }
082                    
083            }
084        }
085        
086        for (int i = 0; i < KNOWN_DIRECTION.length; i++) {
087            if (KNOWN_DIRECTION[i].equals(pDirection)) {
088                return null;
089            }
090        }
091        
092        return UNKNOWN_DIRECTION;
093    }//getExit()
094    
095    /**
096     * Donne la porte en fonction de la direction
097     * @param  pDirection String
098     * @return Door la porte
099     */
100    public Object getDoor(String pDirection) {
101        for (int i = 0; i < KNOWN_DIRECTION.length; i++) {
102            if (KNOWN_DIRECTION[i].equals(pDirection)) {
103                return this.aDoors.get(pDirection);
104            }
105        }
106        
107        return Door.UNKNOWN_DIRECTION_DOOR;
108    }//getDoor()
109    
110    /**
111     * Met les sorties possibles à une room
112     * @param  pDirection String
113     * @param pNeighbor Room (dans quelle direction est la salle)
114     * @param pDoor Door
115     */
116    public void setExits(final String pDirection, final Room pNeighbor, final Door pDoor) {
117        this.aExits.put(pDirection, pNeighbor);
118        this.aDoors.put(pDirection, pDoor); 
119    }//setExits()
120    
121    /**
122     * Renvoie un string avec toutes les sorties
123     * @return   String de toutes les sorties possibles dans la position du joueur
124     */
125    public String getExitString() {
126        StringBuilder vText = new StringBuilder("Exits:");
127        Set<String> keys = this.aExits.keySet();
128        for (String vExits : keys) {
129            vText.append(" " + vExits);
130        }
131        return vText.toString();
132    }//getExitString()
133    
134    /**
135     * Renvoie le nom de l'image
136     * @return String Nom de l'image
137     */
138    public String getImageName() {
139        return this.aImageName;
140    }
141    
142    /**
143     * Ajoute un Item à la room
144     * @param pItem Item
145     */
146    public void addItem(final Item pItem) {
147        this.aItems.addItem(pItem); //String en lower case comme ça plus facile à check pour le joueur
148    }//setItem()
149    
150    /**
151     * Retourne l'item de la room grâce à son nom
152     * @param pName de l'item
153     * @return Item
154     */
155    public Item getItem(final String pName) {
156        return this.aItems.getItem(pName.toLowerCase());
157    }//getItem()
158    
159    /**
160     * Enlève l'item de la room
161     * @param pName de l'item
162     */
163    public void removeItem(final String pName) {
164        this.aItems.removeItem(pName.toLowerCase());
165    }//removeItem()
166    
167    /**
168     * Donne la direction de la salle donnée en paramètre
169     * @param pRoom Room
170     * @return String direction
171     */
172    public String getDirection(final Room pRoom) {
173        for (String vExit : this.aExits.keySet()) {
174            if (this.aExits.get(vExit) == pRoom) {
175                return vExit;
176            }
177        }
178        return null;
179    }//getDirection()
180    
181    /**
182     * Getter si la salle est un transporter
183     * @return boolean false
184     */
185    public boolean isTransporter() {
186        return false;
187    }
188    
189    public String getStringChar() {
190        if (this.aCharacter != null) {
191            return "\n" + this.aCharacter.getDialogue();
192        } else {
193            return "";
194        }
195    }
196    
197} // Room
