001package pkg_rooms;
002
003import java.util.ArrayList;
004import java.util.Random;
005
006/**
007 * Classe qui donne une room al√©atoire.
008 *
009 * @author Luca
010 * @version 24/05/2022
011 */
012public class RoomRandomizer
013{
014    private ArrayList<Room> aRooms;
015    private Random aRandom;
016    private int aAlea;
017    
018    /**
019     * Constructor for objects of class RoomRandomizer
020     * @param pRooms ArrayList de Room
021     */
022    public RoomRandomizer(final ArrayList<Room> pRooms) {
023        this.aRooms = pRooms;
024        this.aRandom = new Random();
025        this.aAlea = -1;
026    }
027
028    /**
029     * Renvoie une salle pseudorandom
030     * @return Room
031     */
032    public Room findRandomRoom() {
033        if (this.aAlea == -1) {
034            return this.aRooms.get(aRandom.nextInt(aRooms.size()));
035        } else {
036            return this.aRooms.get(this.aAlea);
037        }
038    }
039    
040    /**
041     * Setter alea
042     * @param pNum int
043     */
044    public void setAlea(final int pNum) {
045        this.aAlea = pNum;
046    }
047}
