001import java.util.ArrayList;
002
003/**
004 * Classe de Transporter Room
005 *
006 * @author Luca
007 * @version 24/05/2022
008 */
009public class TransporterRoom extends Room
010{
011    private RoomRandomizer aRoomR;
012    
013    /**
014     * Constructeur d'une room
015     * @param pDescription String
016     * @param pImage String
017     * @param pRooms ArrayList de Room
018     */
019    public TransporterRoom(final String pDescription, final String pImage, final ArrayList<Room> pRooms) {
020        super(pDescription, pImage);
021        this.aRoomR = new RoomRandomizer(pRooms); 
022    }
023
024    /**
025     * renvoie la sortie qui est donnée par RoomRandomizer
026     * @param pDirection String
027     * @return Room
028     */
029    @Override
030    public Room getExit(final String pDirection) {
031        return this.aRoomR.findRandomRoom();
032    }
033    
034    /**
035     * renvoie si la room est un transporter 
036     * @return boolean vrai
037     */
038    @Override
039    public boolean isTransporter() {
040        return true;
041    }
042    
043    /**
044     * met l'alea à une salle précise
045     * @param pNum int de l'alea donc de la room de l'ArrayList
046     */
047    public void setAlea(final int pNum) {
048        this.aRoomR.setAlea(pNum);
049    }
050}
