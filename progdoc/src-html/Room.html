001import java.util.HashMap;
002import java.util.Set;
003
004/**
005 * Classe qui permet de créer et gérer les salles du jeu
006 *
007 * @author PALAYSI Luca
008 * @version 24/05/2022
009 */
010public class Room
011{
012    private String aDescription;
013    private HashMap<String, Room> aExits;
014    private HashMap<String, Door> aDoors;
015    public static final Room UNKNOWN_DIRECTION = new Room( "nowhere", "" );
016    public static final Room NO_MOVE_LEFT = new Room( "nomove", "GameOver.jpg" );
017    public static final String[] KNOWN_DIRECTION = {"north", "east", "south", "west", "up", "down"};
018    public static final Room LOCKED_DOOR = new Room( "lockeddoor", "" );
019    private String aImageName;
020    private ItemList aItems;
021
022    /**
023     * Constructeur d'une room
024     * @param pDescription String
025     * @param pImage String
026     */
027    public Room(final String pDescription, final String pImage) {
028        this.aDescription = pDescription;
029        this.aExits = new HashMap<String, Room>();
030        this.aDoors = new HashMap<String, Door>();
031        this.aImageName = pImage;
032        this.aItems = new ItemList();
033    }// Room()
034    
035    /**
036     * Accesseur de la description
037     * @return shortDescription
038     */
039    public String getShortDescription() {
040        return this.aDescription;
041    }//getDescription()
042    
043    /**
044     * Accesseur de la description avec la sortie
045     * @return String longDescription
046     */
047    public String getLongDescription() {
048        return "You are " + this.aDescription + "\n" + this.aItems.getItemString() + "\n" + this.getExitString();
049    }//getLongDescription()
050    
051    /**
052     * Accesseur sorties
053     * @param  pDirection String de la direction souhaitée
054     * @return  Room  La salle de la direction, null s'il n'y pas de salle, unknown direction si la direction est inconnue
055     */
056    public Room getExit(final String pDirection) {
057        for (int i = 0; i < KNOWN_DIRECTION.length; i++) {
058            if (KNOWN_DIRECTION[i].equals(pDirection)) {
059                if (this.aDoors.get(pDirection) != null) {
060                    if (!this.aDoors.get(pDirection).isLocked())
061                        return this.aExits.get(pDirection);
062                    else 
063                        return LOCKED_DOOR;
064                }
065                    
066            }
067        }
068        
069        return UNKNOWN_DIRECTION;
070    }//getExit()
071    
072    /**
073     * Donne la porte en fonction de la direction
074     * @param  pDirection String
075     * @return Door la porte
076     */
077    public Object getDoor(String pDirection) {
078        for (int i = 0; i < KNOWN_DIRECTION.length; i++) {
079            if (KNOWN_DIRECTION[i].equals(pDirection)) {
080                return this.aDoors.get(pDirection);
081            }
082        }
083        
084        return Door.UNKNOWN_DIRECTION_DOOR;
085    }//getDoor()
086    
087    /**
088     * Met les sorties possibles à une room
089     * @param  pDirection String
090     * @param pNeighbor Room (dans quelle direction est la salle)
091     * @param pDoor Door
092     */
093    public void setExits(final String pDirection, final Room pNeighbor, final Door pDoor) {
094        this.aExits.put(pDirection, pNeighbor);
095        this.aDoors.put(pDirection, pDoor); 
096    }//setExits()
097    
098    /**
099     * Renvoie un string avec toutes les sorties
100     * @return   String de toutes les sorties possibles dans la position du joueur
101     */
102    public String getExitString() {
103        StringBuilder vText = new StringBuilder("Exits:");
104        Set<String> keys = this.aExits.keySet();
105        for (String vExits : keys) {
106            vText.append(" " + vExits);
107        }
108        return vText.toString();
109    }//getExitString()
110    
111    /**
112     * Renvoie le nom de l'image
113     * @return String Nom de l'image
114     */
115    public String getImageName() {
116        return this.aImageName;
117    }
118    
119    /**
120     * Ajoute un Item à la room
121     * @param pItem Item
122     */
123    public void addItem(final Item pItem) {
124        this.aItems.addItem(pItem); //String en lower case comme ça plus facile à check pour le joueur
125    }//setItem()
126    
127    /**
128     * Retourne l'item de la room grâce à son nom
129     * @param pName de l'item
130     * @return Item
131     */
132    public Item getItem(final String pName) {
133        return this.aItems.getItem(pName.toLowerCase());
134    }//getItem()
135    
136    /**
137     * Enlève l'item de la room
138     * @param pName de l'item
139     */
140    public void removeItem(final String pName) {
141        this.aItems.removeItem(pName.toLowerCase());
142    }//removeItem()
143    
144    /**
145     * Donne la direction de la salle donnée en paramètre
146     * @param pRoom Room
147     * @return String direction
148     */
149    public String getDirection(final Room pRoom) {
150        for (String vExit : this.aExits.keySet()) {
151            if (this.aExits.get(vExit) == pRoom) {
152                return vExit;
153            }
154        }
155        return null;
156    }//getDirection()
157    
158    /**
159     * Getter si la salle est un transporter
160     * @return boolean false
161     */
162    public boolean isTransporter() {
163        return false;
164    }
165} // Room
